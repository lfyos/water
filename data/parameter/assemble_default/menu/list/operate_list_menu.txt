function(component_name,component_id,render)
{
	render.component_event_processor[component_id]=
	{
		first_show_list_flag	:	true,
		pickupmouseselect		:	function(event,component_id,render)
		{
			var display_ep,ep=render.component_event_processor[component_id];
			var function_id=Math.floor((1.0-ep.y)*19.0);
			switch(function_id){
			case 0:
				return;
			case 1:
				render.system_call_processor.collector_list_create_list(true);
				break;
			case 2:
				render.system_call_processor.collector_list_create_list(false);
				break;
			case 3:
				render.system_call_processor.collector_list_create_list_by_display_target();
				break;
			case 4:
				render.system_call_processor.collector_list_create_list_by_selection_target();
				break;
			case 5:
				render.system_call_processor.collector_list_create_list_by_all_target();
				break;
			case 6:
				return;
			case 7:
				display_ep=render.component_event_processor[
					render.get_component_object_by_component_name("列表功能显示组件").component_id];
				display_ep.collector_stack_version=0;
				display_ep.list_type="simple";
				render.system_call_processor.set_menu_show("列表功能显示组件",
					ep.first_show_list_flag?[["x0",0.75],["y0",-1.0]]:[]);
				render.system_call_processor.set_menu_hide("列表功能操作组件");
				ep.first_show_list_flag=false;
				break;
			case 8:
				display_ep=render.component_event_processor[
					render.get_component_object_by_component_name("列表功能显示组件").component_id];
				display_ep.collector_stack_version=0;
				display_ep.list_type="complex";
				render.system_call_processor.set_menu_show("列表功能显示组件",
					ep.first_show_list_flag?[["x0",0.75],["y0",-1.0]]:[]);
				render.system_call_processor.set_menu_hide("列表功能操作组件");
				ep.first_show_list_flag=false;
				break;
			case 9:
				display_ep=render.component_event_processor[
					render.get_component_object_by_component_name("列表功能显示组件").component_id];
				display_ep.collector_stack_version=0;
				display_ep.list_type="collector";
				render.system_call_processor.set_menu_show("列表功能显示组件",
					ep.first_show_list_flag?[["x0",0.75],["y0",-1.0]]:[]);
				render.system_call_processor.set_menu_hide("列表功能操作组件");
				ep.first_show_list_flag=false;
				break;
			case 10:
				return;
			case 11:
				render.system_call_processor.collector_list_clear();
				break;
			case 12:
				render.system_call_processor.collector_list_save();
				break;
			case 13:
				render.system_call_processor.collector_list_load();
				break;
			case 14:
				window.open(render.system_call_processor.collector_list_upload_webpage("chinese:上传杆件列表"),"_blank");
				break;
			case 15:
				window.open(render.system_call_processor.collector_list_download_url(),"_blank");
				break;
			case 16:
				window.open(render.collector_loader.collector_url(
					confirm("简单列表(YES/NO)"),confirm("单一列表(YES/NO)"),confirm("位置信息(YES/NO)")),"_blank");
				break;
			case 17:
				window.open(render.system_call_processor.collector_list_jason_component_url(ep.x<=0.5),"_blank");
				break;
			default:
				render.system_call_processor.set_menu_show("交互界面主菜单");
				function_id=18;
				break;
			}
			ep.hightlight=[0,1.0-function_id/19.0-1.0/19.0,		1.0,1.0-function_id/19.0];
		}
	}
}
