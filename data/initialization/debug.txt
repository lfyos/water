				function(component_name,component_id,render)
				{
					render.user_call_processor.canvas_scene_url="";
					
					render.user_call_processor.resize=function()
					{
						var part_list;
						if((part_list=document.getElementById("my_part_list"))!=null)
							if(!(part_list.hidden)){
								render.user_call_processor.turnon_control_panel(part_list.src);
								return;
							}
						render.user_call_processor.turnoff_control_panel();
					}
					render.user_call_processor.reload_control_panel=function()
					{
						var part_list;
						if((part_list=document.getElementById("my_part_list"))!=null)
							part_list.src=render.user_call_processor.canvas_scene_url;
					}
					render.user_call_processor.turnon_control_panel=function(my_canvas_scene_url)
					{
						render.user_call_processor.canvas_scene_url=my_canvas_scene_url;
						
						var width	=window.innerWidth *0.975;
						var height	=window.innerHeight*0.925;
						if(width>=height){
							render.canvas.width =width*0.70;
							render.canvas.height=height;
							var part_list;
							if((part_list=document.getElementById("my_part_list"))!=null){
								part_list.width	=width*0.30;
								part_list.height=height;
								part_list.removeAttribute("hidden");
								part_list.src=render.user_call_processor.canvas_scene_url;
							}
						}else{
							render.canvas.width =width;
							render.canvas.height=width;
									
							var part_list;
							if((part_list=document.getElementById("my_part_list"))!=null){
								part_list.width	=width;
								part_list.height=height-width;
									
								part_list.removeAttribute("hidden");

								part_list.src=render.user_call_processor.canvas_scene_url;
							}
						}
					}
					render.user_call_processor.turnoff_control_panel=function()
					{
						var width			=window.innerWidth *0.975;
						var height			=window.innerHeight*0.925;
				
						render.canvas.width=width;
						render.canvas.height=height;
							
						var part_list;
						if((part_list=document.getElementById("my_part_list"))!=null){
							part_list.setAttribute("hidden",true);
							part_list.src="";
						}
					}
					render.system_event_processor.pickupcontextmenu=function(event,pickup_component_id,render)
					{
						var p=render.get_component_to_id_object("movement_manager");
						render.user_call_processor.turnon_control_panel(
							render.component_call_processor[p.component_id].mount_extract_url());
						return false;
					};
					render.system_event_processor.keydown=function(event,render)
					{
						switch(event.keyCode){
						case 27://Esc
							render.system_call_processor.select_component("clear");
							break;
						case 13://Enter
							if(render.pickup.component_id>=0)
								render.system_call_processor.select_component(
									(event.shiftKey||event.ctrlKey||event.altKey)?"switch":"component",
									[["component_id",render.pickup.component_id]]);
							break;
						case 37://left arrow
						case 39://right arrow
						{
							var my_component_id=render.get_component_to_id_object("movement_manager").component_id;
							var mcp=render.component_call_processor[my_component_id];
							
							if(event.keyCode==37)
								mcp.mount_backward(1000);
							else
								mcp.mount_forward(1000);
							
							render.user_call_processor.turnon_control_panel(
								render.system_call_processor.collector_list_webpage_url("simple"));
							break;
						}
						case 38://up arrow
							render.system_call_processor.select_component("parent");
							break;
						case 40://down arrow
							render.system_call_processor.select_component("child");
							break;
						case 32://space
							var my_component_id=render.get_component_to_id_object("movement_manager").component_id;
							var mcp=render.component_call_processor[my_component_id];
							mcp.mount_stop();
							break;
						case 90://z
						case 88://x
						case 67://c
						case 86://v
							var my_function_id;
							switch(event.keyCode){
							default:
							case 90:	my_function_id=000;	break;	//z
							case 88:	my_function_id=100;	break;	//x	
							case 67:	my_function_id=200;	break;	//c
							case 86:	my_function_id=300;	break;	//v
							}
							render.event_component.mouse.	component_name	="mouse_modify_location_component";
							render.event_component.keyboard.component_name	="mouse_modify_location_component";
							render.event_component.touch.	component_name	="mouse_modify_location_component";
							render.event_component.mouse.	function_id		=my_function_id;
							render.event_component.keyboard.function_id		=my_function_id;
							render.event_component.touch.	function_id		=my_function_id;
							render.set_event_component("mouse_modify_location_component");
							break;
						case 66://b
							render.event_component.mouse.	component_name	="component_selection_component";
							render.event_component.keyboard.component_name	="component_selection_component";
							render.event_component.mouse.	function_id		=0;
							render.event_component.keyboard.function_id		=0;
							render.set_event_component("component_selection_component");
							break;
						case 78://n
							render.event_component.mouse.	component_name	="tag_controller";
							render.event_component.keyboard.component_name	="tag_controller";
							render.event_component.mouse.	function_id		=function_id;
							render.event_component.keyboard.function_id		=function_id;
							render.set_event_component("tag_controller");
							break;
						case 77://m
							render.event_component.mouse.	component_name	="text_token_component";
							render.event_component.keyboard.component_name	="text_token_component";
							render.event_component.mouse.	function_id		=0;
							render.event_component.keyboard.function_id		=0;
							render.set_event_component("text_token_component");
							break;
						case 112://F1
							render.user_event_processor.set_all_transparant_except_selection();
							break;
						case 113://F2
							render.user_event_processor.set_all_untransparant()
							break;
						case 114://F3
							window.open(render.system_call_processor.component_check_url(),"_blank");
							break;
						case 115://F4
							break;
						case 116://F5
							break;
						case 117://F6
							window.open(
								render.create_component_request_string(
									"find_visible_component","all",[["operation","file"]]),
								"_blank");
							break;
						case 118://F7
							var precision;
							if((precision=prompt("输入精度","1"))==null)
								break;
							precision=parseFloat(precision);
							render.call_server_component("find_visible_component","all",
								[	["operation","start"],
									["width",	Math.round(render.canvas.width *precision)],
									["height",	Math.round(render.canvas.height*precision)]
								],
								function(response_data,render)
								{
									var my_component_id=render.get_component_to_id_object("find_visible_component").component_id;
									render.component_event_processor[my_component_id].clear_triangle(my_component_id);
								});
							break;
						case 119://F8
							var my_component_id=render.get_component_to_id_object("find_visible_component").component_id;
							render.component_event_processor[my_component_id].caculate_triangle(my_component_id);
							break;
						case 120://F9
							break;
						case 121://F10
							break;
						case 122://F11
							break;
						case 123://F12
							render.system_call_processor.set_menu_show_me_only("交互界面主菜单");
							break;
						};
						return false;
					};
				}
