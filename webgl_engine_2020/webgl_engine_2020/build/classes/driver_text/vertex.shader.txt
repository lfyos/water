
uniform mat4 projection_matrix;
uniform vec2 width_heigh;
uniform int direction_id;

in vec4 vertex_location,vertex_texture;

out vec2 texture_loca;

void main(void)
{
	if(direction_id==0)
		texture_loca=vec2(    vertex_texture.x,    vertex_texture.y);
	else if(direction_id==1)
		texture_loca=vec2(1.0-vertex_texture.x,    vertex_texture.y);
	else if(direction_id==2)
		texture_loca=vec2(    vertex_texture.x,1.0-vertex_texture.y);
	else
		texture_loca=vec2(1.0-vertex_texture.x,1.0-vertex_texture.y);

	gl_Position=projection_matrix*(component_info.model_matrix*vec4(vertex_location.xy*width_heigh,0.0,1.0));
}
